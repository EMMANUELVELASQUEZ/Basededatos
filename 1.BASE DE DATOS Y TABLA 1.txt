1.BASE DE DATOS Y TABLA 1:

CREATE TABLE DEPARTAMENTOS (
 CLAVEDEPTO CHAR(3) NOT NULL,
 DEPTOSUP CHAR(3) NOT NULL,
 NOMDEPTO VARCHAR(36) UNIQUE NOT NULL,
JEFE CHAR(6),
PRIMARI KEY(CLAVEDEPTO)
);
 
2.Segunda tabla:

CREATE TABLE EMPLEADOS (
    MATRICULA CHAR(6) NOT NULL,
    NUMDEPTO CHAR(3) NOT NULL,
    NOMBRE VARCHAR(12) NOT NULL,
    APELLIDOS VARCHAR (15) NOT NULL,
    EXTENSION CHAR(6) NOT NULL,
    FECHAING DATE,
    PUESTO CHAR(8) NOT NULL,
    NIVELEST SMALLINT,
    GENERO CHAR(1),
    FECHANAC DATE,
    SALARIO DECIMAL(9,2),
    COMISION DECIMAL(9,2),
PRIMARY KEY(MATRICULA),
FOREIGN KEY (NUMDEPTO) REFERENCES DEPARTAMENTOS(CLAVEDEPTO)
ON DELETE CASCADE ON UPDATE CASCADE
);
